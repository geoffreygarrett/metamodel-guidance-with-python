18static const real_T rt_ODE3_A[3] = {



19    1.0/2.0, 3.0/4.0, 1.0



20};



21



22static const real_T rt_ODE3_B[3][3] = {



23    { 1.0/2.0,     0.0,     0.0 },



24    {     0.0, 3.0/4.0,     0.0 },



25    { 2.0/9.0, 1.0/3.0, 4.0/9.0 }



26};

29typedef struct IntgData_tag {



30    real_T *y;



31    real_T *f[3];



32} IntgData;



33



34



35#ifndef RT_MALLOC



36  /* statically declare data */



37  static real_T   rt_ODE3_Y[NCSTATES];



38  static real_T   rt_ODE3_F[3][NCSTATES];



39  static IntgData rt_ODE3_IntgData = {rt_ODE3_Y,



40                                      {rt_ODE3_F[0],rt_ODE3_F[1],rt_ODE3_F[2]}};



41



42void rt_ODECreateIntegrationData(RTWSolverInfo *si)



43  {



44      rtsiSetSolverData(si,(void *)&rt_ODE3_IntgData);



45      rtsiSetSolverName(si,"ode3");



46  }



47#else



48  /* dynamically allocated data */



49



50  void rt_ODECreateIntegrationData(RTWSolverInfo *si)



51  {



52      IntgData *id = (IntgData *) malloc(sizeof(IntgData));



53      if(id == NULL) {



54          rtsiSetErrorStatus(si, RT_MEMORY_ALLOCATION_ERROR);



55          return;



56      }



57      



58      id->y = (real_T *) malloc(4*rtsiGetNumContStates(si) * sizeof(real_T));



59      if(id->y == NULL) {



60          rtsiSetErrorStatus(si, RT_MEMORY_ALLOCATION_ERROR);



61          return;



62      }



63      id->f[0] = id->y + rtsiGetNumContStates(si);



64      id->f[1] = id->f[0] + rtsiGetNumContStates(si);



65      id->f[2] = id->f[1] + rtsiGetNumContStates(si);



66      



67      rtsiSetSolverData(si, (void *)id);



68      rtsiSetSolverName(si,"ode3");



69  }



70



71  void rt_ODEDestroyIntegrationData(RTWSolverInfo *si)



72  {



73      IntgData *id = rtsiGetSolverData(si);



74      



75      if (id != NULL) {



76          if (id->y != NULL) {



77              free(id->y);



78          }



79          free(id);



80          rtsiSetSolverData(si, NULL);



81      }



82  }



83#endif

85void rt_ODEUpdateContinuousStates(RTWSolverInfo *si)



86{



87    time_T    t          = rtsiGetT(si);



88    time_T    tnew       = rtsiGetSolverStopTime(si);



89    time_T    h          = rtsiGetStepSize(si);



90    real_T    *x         = rtsiGetContStates(si);



91    IntgData  *id        = rtsiGetSolverData(si);



92    real_T    *y         = id->y;



93    real_T    *f0        = id->f[0];



94    real_T    *f1        = id->f[1];



95    real_T    *f2        = id->f[2];



96    real_T    hB[3];



97    int_T     i;



98



99#ifdef NCSTATES



100    int_T     nXc        = NCSTATES;



101#else



102    int_T     nXc        = rtsiGetNumContStates(si);



103#endif



104



105    rtsiSetSimTimeStep(si,MINOR_TIME_STEP);



106



107    /* Save the state values at time t in y, we'll use x as ynew. */



108    (void)memcpy(y, x, nXc*sizeof(real_T));



109



110    /* Assumes that rtsiSetT and ModelOutputs are up-to-date */



111    /* f0 = f(t,y) */



112    rtsiSetdX(si, f0);



113    DERIVATIVES(si);



114



115    /* f(:,2) = feval(odefile, t + hA(1), y + f*hB(:,1), args(:)(*)); */



116    hB[0] = h * rt_ODE3_B[0][0];



117    for (i = 0; i < nXc; i++) {



118	x[i] = y[i] + (f0[i]*hB[0]);



119    }



120    rtsiSetT(si, t + h*rt_ODE3_A[0]);



121    rtsiSetdX(si, f1);



122    OUTPUTS(si,0);



123    DERIVATIVES(si);



124



125    /* f(:,3) = feval(odefile, t + hA(2), y + f*hB(:,2), args(:)(*)); */



126    for (i = 0; i <= 1; i++) hB[i] = h * rt_ODE3_B[1][i];



127    for (i = 0; i < nXc; i++) {



128	x[i] = y[i] + (f0[i]*hB[0] + f1[i]*hB[1]);



129    }



130    rtsiSetT(si, t + h*rt_ODE3_A[1]);



131    rtsiSetdX(si, f2);



132    OUTPUTS(si,0);



133    DERIVATIVES(si);



134



135    /* tnew = t + hA(3);



136       ynew = y + f*hB(:,3); */



137    for (i = 0; i <= 2; i++) hB[i] = h * rt_ODE3_B[2][i];



138    for (i = 0; i < nXc; i++) {



139	x[i] = y[i] + (f0[i]*hB[0] + f1[i]*hB[1] + f2[i]*hB[2]);



140    }



141    rtsiSetT(si, tnew);



142



143    PROJECTION(si);



144    REDUCTION(si);



145



146    rtsiSetSimTimeStep(si,MAJOR_TIME_STEP);



147}
